Ejemplo práctico en C#

Clases, herencia y polimorfismo modelando una biblioteca.

Escenario

Modelamos una biblioteca con una clase base Publicacion y dos derivadas: Libro y Revista. Mostramos polimorfismo sobrescribiendo GetDescripcion().

// Ejemplo en C#
using System;
using System.Collections.Generic;

abstract class Publicacion
{
    public string Titulo { get; }
    public string Autor { get; }

    protected Publicacion(string titulo, string autor)
    {
        Titulo = titulo;
        Autor = autor;
    }

    public abstract string GetDescripcion();
}

class Libro : Publicacion
{
    public int Paginas { get; }

    public Libro(string titulo, string autor, int paginas) : base(titulo, autor)
    {
        Paginas = paginas;
    }

    public override string GetDescripcion()
    {
        return $"Libro: '{Titulo}' de {Autor} ({Paginas} páginas)";
    }
}

class Revista : Publicacion
{
    public int Numero { get; }

    public Revista(string titulo, string autor, int numero) : base(titulo, autor)
    {
        Numero = numero;
    }

    public override string GetDescripcion()
    {
        return $"Revista: '{Titulo}' edición #{Numero}, editor {Autor}";
    }
}

class Programa
{
    static void Main()
    {
        var catalogo = new List
        {
            new Libro("El Quijote", "Cervantes", 863),
            new Revista("TechToday", "Acme Media", 142)
        };

        foreach (var pub in catalogo)
        {
            Console.WriteLine(pub.GetDescripcion());
        }
    }
}
